import pygame
import sys

# Initialize Pygame
pygame.init()

# Set up the display
screen = pygame.display.set_mode((800, 600))
pygame.display.set_caption("Simple Pygame Example")

# Define colors
WHITE = (255, 255, 255)
RED = (255, 0, 0)

# Set up the rectangle
rect_width = 50
rect_height = 50
rect_x = 0
rect_y = 275
rect_speed = 5

# Game loop
running = True
while running:
    # Handle events
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False
    
    # Move the rectangle
    rect_x += rect_speed
    if rect_x > 800:
        rect_x = 0  # Reset position when it moves off screen
    
    # Fill the screen with white
    screen.fill(WHITE)
    
    # Draw the rectangle
    pygame.draw.rect(screen, RED, (rect_x, rect_y, rect_width, rect_height))
    
    # Update the display
    pygame.display.flip()
    
    # Control the frame rate
    pygame.time.Clock().tick(60)

# Quit Pygame
pygame.quit()
sys.exit()
